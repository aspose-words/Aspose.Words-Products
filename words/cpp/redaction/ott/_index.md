// Load OTT document
System::SharedPtr<Document> ott = System::MakeObject<Document>(u"Document.ott");

// Create find and replace options
System::SharedPtr<FindReplaceOptions> options = System::MakeObject<FindReplaceOptions>();

// Find and replace the text
ott->get_Range()->Replace(System::MakeObject<System::Text::RegularExpressions::Regex>(u"[s|m]ad"), u"bad", options);

// Save the updated document 
ott->Save(u"updated.ott");
