// Load MD document
System::SharedPtr<Document> md = System::MakeObject<Document>(u"Document.md");

// Create find and replace options
System::SharedPtr<FindReplaceOptions> options = System::MakeObject<FindReplaceOptions>();

// Find and replace the text
md->get_Range()->Replace(System::MakeObject<System::Text::RegularExpressions::Regex>(u"[s|m]ad"), u"bad", options);

// Save the updated document 
md->Save(u"updated.md");
