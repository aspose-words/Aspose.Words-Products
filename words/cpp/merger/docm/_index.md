// Load docm documents to be merged
System::SharedPtr<Aspose::Words::Document> doc1 = System::MakeObject<Aspose::Words::Document>(u"SourceFile1.docm");
System::SharedPtr<Aspose::Words::Document> doc2 = System::MakeObject<Aspose::Words::Document>(u"SourceFile2.docm");

// Merge documents keeping source file formatting
doc1->AppendDocument(doc2, Aspose::Words::ImportFormatMode::KeepSourceFormatting);

// Save merged document as DOCM file
doc1->Save(u"merged-doc-out.docm", Aspose::Words::SaveFormat::Docm);
