// Load rtf documents to be merged
System::SharedPtr<Aspose::Words::Document> doc1 = System::MakeObject<Aspose::Words::Document>(u"SourceFile1.rtf");
System::SharedPtr<Aspose::Words::Document> doc2 = System::MakeObject<Aspose::Words::Document>(u"SourceFile2.rtf");

// Merge documents keeping source file formatting
doc1->AppendDocument(doc2, Aspose::Words::ImportFormatMode::KeepSourceFormatting);

// Save merged document as RTF file
doc1->Save(u"merged-doc-out.rtf", Aspose::Words::SaveFormat::Rtf);
