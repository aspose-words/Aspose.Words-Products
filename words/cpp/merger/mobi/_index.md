// Load mobi documents to be merged
System::SharedPtr<Aspose::Words::Document> doc1 = System::MakeObject<Aspose::Words::Document>(u"SourceFile1.mobi");
System::SharedPtr<Aspose::Words::Document> doc2 = System::MakeObject<Aspose::Words::Document>(u"SourceFile2.mobi");

// Merge documents keeping source file formatting
doc1->AppendDocument(doc2, Aspose::Words::ImportFormatMode::KeepSourceFormatting);

// Save merged document as MOBI file
doc1->Save(u"merged-doc-out.mobi", Aspose::Words::SaveFormat::Mobi);
