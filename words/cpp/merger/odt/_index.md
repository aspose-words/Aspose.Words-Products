// Load odt documents to be merged
System::SharedPtr<Aspose::Words::Document> doc1 = System::MakeObject<Aspose::Words::Document>(u"SourceFile1.odt");
System::SharedPtr<Aspose::Words::Document> doc2 = System::MakeObject<Aspose::Words::Document>(u"SourceFile2.odt");

// Merge documents keeping source file formatting
