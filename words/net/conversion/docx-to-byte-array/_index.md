```cs
// Create instance and load the DOCX file from disk.
Document docx = new Document("SourceSample.docx");

// Create a new memory stream.
MemoryStream outStream = new MemoryStream();
// Save the document to stream.
docx.Save(outStream, SaveFormat.Docx);

// Convert the document to byte array.
byte[] docxBytes = outStream.ToArray();

// Bytes are ready to be processed/stored.

// In order to convert byte array to DOCX
// Now reverse the steps to load the bytes back into a document object.
MemoryStream inStream = new MemoryStream(docxBytes);

// Load the stream into a new object.
Document load_docx = new Document(inStream);
// Save the file.
load_docx.Save("output.docx", SaveFormat.Docx); 

```
